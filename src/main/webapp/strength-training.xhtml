// Script para mejorar la visualización de los radio buttons personalizados document.addEventListener('DOMContentLoaded', function() { const options = document.querySelectorAll('.checkbox-option'); options.forEach(option => { const radio = option.querySelector('input[type="radio"]'); const checkmark = option.querySelector('.checkmark'); // Actualizar estado inicial if(radio.checked) { checkmark.style.backgroundColor = '#4da8da'; checkmark.style.borderColor = '#4da8da'; checkmark.innerHTML = ''; } // Manejar cambios radio.addEventListener('change', function() { options.forEach(opt => { const chk = opt.querySelector('.checkmark'); chk.style.backgroundColor = '#fff'; chk.style.borderColor = '#b0c4de'; chk.innerHTML = ''; }); if(this.checked) { checkmark.style.backgroundColor = '#4da8da'; checkmark.style.borderColor = '#4da8da'; checkmark.innerHTML = ''; } }); }); });
